### 1.1 当前架构（v1.0）

```

  app/
    main.py  (1900+ 行，所有功能耦合)
```

```python
async def get_db():
```

```python
import sqlite3
import asyncio
from sqlalchemy import select
from backend.core.database import AsyncSessionLocal
from backend.models import Site, User, Task

async def migrate():
    # 读取 SQLite 数据
    sqlite_conn = sqlite3.connect("/shared/app.db")
    sqlite_conn.row_factory = sqlite3.Row

    rows = sqlite_conn.execute("SELECT * FROM sub_sites").fetchall()

    async with AsyncSessionLocal() as session:
        for row in rows:
            site = Site(
                site_id=row["site_id"],
                name=row["name"] or row["site_id"],
                status=row["status"],
                port=row["port"],
                # org_id/owner_id 需要预先创建默认组织/用户
                org_id="default-org-uuid",
                owner_id="default-user-uuid",
            )
            session.add(site)

        await session.commit()

    sqlite_conn.close()
    print(f"✅ 迁移完成：{len(rows)} 个站点")

if __name__ == "__main__":
    asyncio.run(migrate())
```

```python
from backend.services.site_service import SiteService

class DevelopCodeTask(Task):
    def on_failure(self, exc, task_id, args, kwargs, einfo):
        """任务失败回调"""
        # 更新数据库状态
        # 发送告警通知
        pass

@celery_app.task(
    bind=True,
    base=DevelopCodeTask,
    max_retries=3,
    default_retry_delay=60,  # 重试间隔60秒
)
def develop_code_task(self, task_id: str, site_id: str, provider: str, prompt: str):
    """
    AI 开发代码任务（异步）

    Args:
        task_id: 任务ID
        site_id: 站点ID
        provider: AI Provider (codex/claude/gemini)
        prompt: 用户需求
    """
    try:
        # 1. 更新任务状态为 running
        update_task_status(task_id, TaskStatus.RUNNING)

        # 2. 获取站点信息
        site = get_site(site_id)

        # 3. 构建命令
        cmd = build_provider_command(provider, prompt)

        # 4. 执行命令（日志流式写入）
        result = execute_command_with_logs(
            task_id=task_id,
            command=cmd,
            cwd=site.root_path,
            timeout=1800
        )

        # 5. 更新任务状态为 success
        update_task_status(task_id, TaskStatus.SUCCESS, result=result)

        return {"ok": True, "exit_code": result.exit_code}

    except Exception as exc:
        # 6. 重试逻辑
        if self.request.retries < self.max_retries:
            raise self.retry(exc=exc)

        # 7. 最终失败
        update_task_status(task_id, TaskStatus.FAILED, error=str(exc))
        raise
```

#### 认证实现

```python
# backend/core/security.py
from datetime import datetime, timedelta
from passlib.context import CryptContext
from jose import JWTError, jwt

SECRET_KEY = "your-secret-key-here"  # 生产环境使用环境变量
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 30
REFRESH_TOKEN_EXPIRE_DAYS = 7

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def verify_password(plain_password: str, hashed_password: str) -> bool:
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password: str) -> str:
    return pwd_context.hash(password)

def create_access_token(data: dict, expires_delta: timedelta = None):
    to_encode = data.copy()
    expire = datetime.utcnow() + (expires_delta or timedelta(minutes=15))
    to_encode.update({"exp": expire, "type": "access"})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

def create_refresh_token(data: dict):
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(days=REFRESH_TOKEN_EXPIRE_DAYS)
    to_encode.update({"exp": expire, "type": "refresh"})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

async def get_current_user(
    token: str = Depends(oauth2_scheme),
    db: AsyncSession = Depends(get_db)
) -> User:
    credentials_exception = HTTPException(
        status_code=401,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        user_id: str = payload.get("sub")
        if user_id is None or payload.get("type") != "access":
            raise credentials_exception
    except JWTError:
        raise credentials_exception

    result = await db.execute(select(User).where(User.id == user_id))
    user = result.scalar_one_or_none()
    if user is None:
        raise credentials_exception
    return user
```

#### 认证 API

```python
# backend/api/v2/auth.py
from fastapi import APIRouter, Depends, HTTPException
from fastapi.security import OAuth2PasswordRequestForm

router = APIRouter(prefix="/api/v2/auth", tags=["Auth"])

@router.post("/register")
async def register(
    email: str,
    password: str,
    name: str = None,
    db: AsyncSession = Depends(get_db)
):
    # 检查邮箱是否已存在
    result = await db.execute(select(User).where(User.email == email))
    if result.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="Email already registered")

    # 创建用户
    user = User(
        email=email,
        password_hash=get_password_hash(password),
        name=name,
    )
    db.add(user)

    # 创建默认组织
    org = Organization(
        name=f"{name}'s Organization",
        slug=f"{name.lower().replace(' ', '-')}-{uuid.uuid4().hex[:6]}",
    )
    db.add(org)
    await db.flush()

    # 添加用户到组织（owner角色）
    member = OrganizationMember(org_id=org.id, user_id=user.id, role="owner")
    db.add(member)

    await db.commit()

    return {"ok": True, "user_id": str(user.id)}

@router.post("/login")
async def login(
    form_data: OAuth2PasswordRequestForm = Depends(),
    db: AsyncSession = Depends(get_db)
):
    # 验证用户
    result = await db.execute(select(User).where(User.email == form_data.username))
    user = result.scalar_one_or_none()

    if not user or not verify_password(form_data.password, user.password_hash):
        raise HTTPException(status_code=401, detail="Incorrect email or password")

    # 生成令牌
    access_token = create_access_token(data={"sub": str(user.id)})
    refresh_token = create_refresh_token(data={"sub": str(user.id)})

    # 更新最后登录时间
    user.last_login_at = datetime.utcnow()
    await db.commit()

    return {
        "access_token": access_token,
        "refresh_token": refresh_token,
        "token_type": "bearer",
        "expires_in": ACCESS_TOKEN_EXPIRE_MINUTES * 60,
    }

@router.post("/refresh")
async def refresh_token(
    refresh_token: str,
    db: AsyncSession = Depends(get_db)
):
    try:
        payload = jwt.decode(refresh_token, SECRET_KEY, algorithms=[ALGORITHM])
        if payload.get("type") != "refresh":
            raise HTTPException(status_code=401, detail="Invalid token type")

        user_id = payload.get("sub")
        access_token = create_access_token(data={"sub": user_id})

        return {
            "access_token": access_token,
            "token_type": "bearer",
        }
    except JWTError:
        raise HTTPException(status_code=401, detail="Invalid refresh token")

@router.get("/me")
async def get_current_user_info(
    current_user: User = Depends(get_current_user)
):
    return {
        "id": str(current_user.id),
        "email": current_user.email,
        "name": current_user.name,
        "avatar_url": current_user.avatar_url,
    }
```

```python
# backend/services/version_service.py
class VersionService:
    async def create_snapshot(
        self,
        site_id: str,
        commit_message: str,
        created_by: str,
        db: AsyncSession
    ) -> SiteVersion:
        # 1. 获取站点
        site = await db.get(Site, site_id)

        # 2. 计算版本号
        result = await db.execute(
            select(func.max(SiteVersion.version_number))
            .where(SiteVersion.site_id == site_id)
        )
        last_version = result.scalar() or 0
        new_version_number = last_version + 1

        # 3. 打包站点代码
        snapshot_path = f"/tmp/{site_id}_{new_version_number}.tar.gz"
        tar_command = f"tar -czf {snapshot_path} -C {site.root_path} ."
        subprocess.run(tar_command, shell=True, check=True)

        # 4. 上传到 MinIO
        snapshot_url = await upload_to_minio(
            snapshot_path,
            bucket="site-versions",
            object_name=f"{site_id}/v{new_version_number}.tar.gz"
        )

        # 5. 计算 diff（与上一版本对比）
        diff_summary = {}
        if last_version > 0:
            diff_summary = await self.calculate_diff(site_id, last_version, new_version_number)

        # 6. 创建版本记录
        version = SiteVersion(
            site_id=site.id,
            version_number=new_version_number,
            snapshot_url=snapshot_url,
            commit_message=commit_message,
            diff_summary=diff_summary,
            created_by=created_by,
        )
        db.add(version)
        await db.commit()

        # 7. 清理临时文件
        os.remove(snapshot_path)

        return version

    async def rollback_to_version(
        self,
        site_id: str,
        version_number: int,
        db: AsyncSession
    ):
        # 1. 获取目标版本
        result = await db.execute(
            select(SiteVersion)
            .where(
                SiteVersion.site_id == site_id,
                SiteVersion.version_number == version_number
            )
        )
        target_version = result.scalar_one_or_none()
        if not target_version:
            raise ValueError(f"Version {version_number} not found")

        # 2. 下载快照
        snapshot_path = await download_from_minio(target_version.snapshot_url)

        # 3. 停止站点
        site = await db.get(Site, site_id)
        await stop_site_process(site.site_id)

        # 4. 替换代码
        shutil.rmtree(site.root_path, ignore_errors=True)
        os.makedirs(site.root_path, exist_ok=True)
        subprocess.run(f"tar -xzf {snapshot_path} -C {site.root_path}", shell=True, check=True)

        # 5. 重启站点
        await start_site_process(site.site_id, site.port)

        # 6. 清理临时文件
        os.remove(snapshot_path)
```

```python
# backend/services/container_service.py
class ContainerService:
    async def create_site_container(self, site: Site) -> str:
        """
        为站点创建独立Docker容器

        Returns:
            container_id: 容器ID
        """
        # 1. 生成 Dockerfile
        dockerfile_path = site.root_path / "Dockerfile"
        dockerfile_content = self._generate_dockerfile(site)
        dockerfile_path.write_text(dockerfile_content)

        # 2. 构建镜像
        image_tag = f"nextproject/site-{site.site_id}:latest"
        build_command = [
            "docker", "build",
            "-t", image_tag,
            "-f", str(dockerfile_path),
            str(site.root_path)
        ]
        subprocess.run(build_command, check=True)

        # 3. 启动容器
        run_command = [
            "docker", "run",
            "-d",  # 后台运行
            "--name", f"site-{site.site_id}",
            "--network", "nextproject-network",
            "-p", f"{site.port}:8080",
            "--label", f"site_id={site.site_id}",
            "--label", f"org_id={site.org_id}",
            # 资源限制
            "--memory", "512m",
            "--cpus", "0.5",
            # 环境变量
            "-e", f"SITE_ID={site.site_id}",
            "-e", f"DATABASE_URL={os.getenv('DATABASE_URL')}",
            # 卷挂载（如需持久化）
            "-v", f"{site.root_path}/data:/app/data",
            image_tag
        ]

        result = subprocess.run(run_command, capture_output=True, text=True, check=True)
        container_id = result.stdout.strip()

        # 4. 等待健康检查
        await self._wait_for_healthy(container_id, timeout=30)

        return container_id

    def _generate_dockerfile(self, site: Site) -> str:
        """根据站点技术栈生成 Dockerfile"""
        tech_stack = site.config.get("tech_stack", {})
        backend = tech_stack.get("backend", "fastapi")
        frontend = tech_stack.get("frontend", "vue3")

        if backend == "fastapi" and frontend == "vue3":
            return """
FROM node:20-alpine AS frontend-build
WORKDIR /build
COPY frontend/package*.json ./
RUN npm ci --production
COPY frontend/ ./
RUN npm run build

FROM python:3.12-slim
WORKDIR /app
COPY backend/requirements.txt ./
RUN pip install --no-cache-dir -r requirements.txt
COPY backend/ ./backend
COPY --from=frontend-build /build/dist ./frontend

HEALTHCHECK --interval=10s --timeout=3s --retries=3 \\
  CMD curl -f http://localhost:8080/health || exit 1

EXPOSE 8080
CMD ["uvicorn", "backend.app:app", "--host", "0.0.0.0", "--port", "8080"]
"""
        else:
            # 其他技术栈的 Dockerfile...
            pass

    async def _wait_for_healthy(self, container_id: str, timeout: int = 30):
        """等待容器健康检查通过"""
        import time
        start = time.time()
        while time.time() - start < timeout:
            result = subprocess.run(
                ["docker", "inspect", "--format={{.State.Health.Status}}", container_id],
                capture_output=True,
                text=True
            )
            status = result.stdout.strip()
            if status == "healthy":
                return
            await asyncio.sleep(1)

        raise TimeoutError(f"Container {container_id} did not become healthy in {timeout}s")
```

### 4.2 WebSocket 实时推送

```python
# backend/api/v2/websocket.py
from fastapi import WebSocket, WebSocketDisconnect

class ConnectionManager:
    def __init__(self):
        self.active_connections: Dict[str, List[WebSocket]] = {}

    async def connect(self, task_id: str, websocket: WebSocket):
        await websocket.accept()
        if task_id not in self.active_connections:
            self.active_connections[task_id] = []
        self.active_connections[task_id].append(websocket)

    def disconnect(self, task_id: str, websocket: WebSocket):
        self.active_connections[task_id].remove(websocket)

    async def broadcast(self, task_id: str, message: dict):
        if task_id in self.active_connections:
            for connection in self.active_connections[task_id]:
                await connection.send_json(message)

manager = ConnectionManager()

@app.websocket("/ws/tasks/{task_id}/logs")
async def task_logs_websocket(websocket: WebSocket, task_id: str):
    await manager.connect(task_id, websocket)
    try:
        # 订阅 Redis Pub/Sub（Celery 任务日志）
        pubsub = redis_client.pubsub()
        await pubsub.subscribe(f"task:{task_id}:logs")

        async for message in pubsub.listen():
            if message["type"] == "message":
                await websocket.send_json({
                    "type": "log",
                    "data": message["data"]
                })
    except WebSocketDisconnect:
        manager.disconnect(task_id, websocket)
```

| 风险 | 影响 | 概率 | 缓解措施 |
|------|------|------|----------|
| 数据库迁移数据丢失 | 高 | 低 | 1. 迁移前全量备份<br>2. 灰度迁移（先迁移10%数据验证）<br>3. 保留 SQLite 备份 30 天 |
| Celery 任务积压 | 中 | 中 | 1. 设置任务超时<br>2. 监控队列长度告警<br>3. 自动扩容 Worker |
| 容器资源耗尽 | 高 | 中 | 1. 容器资源限制（CPU/内存配额）<br>2. 自动清理空闲容器<br>3. K8s HPA 自动扩容 |
| AI API 费用超支 | 高 | 高 | 1. 用户配额限制<br>2. 费用实时监控告警<br>3. 缓存相似 prompt |
