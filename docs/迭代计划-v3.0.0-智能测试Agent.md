## 🎯 迭代目标

使用 **pi-mono agent 框架**（@mariozechner/pi-agent-core）构建智能化冒烟测试系统，实现对新生成网站的自动化功能验证，替代当前的 Playwright 脚本化测试。

| 功能 | 当前（v2.0） | 目标（v3.0） |
|------|------------|------------|
| **测试用例生成** | 手动编写脚本 | AI 自动生成 |

| **故障恢复** | 失败即停止 | 自动重试 + 策略调整 |

```typescript
import { AgentTool } from "@mariozechner/pi-agent-core";
import { Type } from "@sinclair/typebox";
import { chromium, Browser, Page } from "playwright";

export const playwrightNavigateTool: AgentTool = {
  name: "playwright_navigate",
  label: "导航到页面",
  description: "打开浏览器并导航到指定URL",

  parameters: Type.Object({
    url: Type.String({ description: "目标URL" }),
    waitUntil: Type.Optional(
      Type.Union([
        Type.Literal("load"),
        Type.Literal("domcontentloaded"),
        Type.Literal("networkidle"),
      ])
    ),
  }),

  execute: async (toolCallId, params, signal, onUpdate) => {
    const { url, waitUntil = "networkidle" } = params;

    // 从全局上下文获取或创建 browser
    let browser = globalThis._testBrowser as Browser | undefined;
    if (!browser) {
      browser = await chromium.launch({ headless: true });
      globalThis._testBrowser = browser;
    }

    const page = await browser.newPage();
    globalThis._testPage = page;

    onUpdate?.({
      content: [{ type: "text", text: `正在导航到 ${url}...` }],
      details: {},
    });

    await page.goto(url, { waitUntil });

    const title = await page.title();
    const finalUrl = page.url();

    return {
      content: [
        {
          type: "text",
          text: `成功导航到页面\n标题: ${title}\nURL: ${finalUrl}`,
        },
      ],
      details: {
        title,
        url: finalUrl,
        viewport: await page.viewportSize(),
      },
    };
  },
};
```

```typescript
// backend/services/test_agent/tools/analyze_page.ts

export const analyzePageTool: AgentTool = {
  name: "analyze_page",
  label: "分析页面结构",
  description: "分析当前页面的DOM结构、交互元素、功能模块",

  parameters: Type.Object({
    depth: Type.Optional(Type.Number({ description: "分析深度 (1-3)", minimum: 1, maximum: 3 })),
  }),

  execute: async (toolCallId, params, signal, onUpdate) => {
    const page = globalThis._testPage as Page;
    if (!page) {
      throw new Error("请先使用 playwright_navigate 打开页面");
    }

    const depth = params.depth || 2;

    // 提取页面结构
    const pageStructure = await page.evaluate((depth) => {
      const result: any = {
        title: document.title,
        url: window.location.href,
        elements: {
          buttons: [],
          links: [],
          forms: [],
          inputs: [],
        },
        sections: [],
      };

      // 提取按钮
      document.querySelectorAll("button, [role='button']").forEach((btn) => {
        result.elements.buttons.push({
          text: btn.textContent?.trim() || "",
          id: btn.id,
          class: btn.className,
          visible: (btn as HTMLElement).offsetParent !== null,
        });
      });

      // 提取链接
      document.querySelectorAll("a").forEach((link) => {
        result.elements.links.push({
          text: link.textContent?.trim() || "",
          href: link.href,
          id: link.id,
        });
      });

      // 提取表单
      document.querySelectorAll("form").forEach((form) => {
        const fields: any[] = [];
        form.querySelectorAll("input, textarea, select").forEach((field) => {
          fields.push({
            name: (field as HTMLInputElement).name,
            type: (field as HTMLInputElement).type,
            required: (field as HTMLInputElement).required,
            placeholder: (field as HTMLInputElement).placeholder,
          });
        });

        result.elements.forms.push({
          id: form.id,
          action: form.action,
          method: form.method,
          fields,
        });
      });

      // 提取主要内容区域（启发式）
      if (depth >= 2) {
        document.querySelectorAll("section, article, main, [role='main']").forEach((section) => {
          const heading = section.querySelector("h1, h2, h3")?.textContent?.trim();
          if (heading) {
            result.sections.push({
              heading,
              id: section.id,
              class: section.className,
            });
          }
        });
      }

      return result;
    }, depth);

    // AI 分析：识别功能模块
    const analysis = {
      ...pageStructure,
      insights: {
        hasAuth: pageStructure.elements.forms.some((f: any) =>
          f.fields.some((field: any) => field.type === "password")
        ),
        hasSearch: pageStructure.elements.inputs.some((i: any) =>
          i.type === "search" || i.placeholder?.includes("搜索")
        ),
        isInteractive: pageStructure.elements.buttons.length > 0,
        complexity: pageStructure.elements.buttons.length +
          pageStructure.elements.forms.length * 2,
      },
    };

    return {
      content: [
        {
          type: "text",
          text: `页面分析完成：
- 按钮: ${pageStructure.elements.buttons.length} 个
- 链接: ${pageStructure.elements.links.length} 个
- 表单: ${pageStructure.elements.forms.length} 个
- 功能模块: ${pageStructure.sections.length} 个

${analysis.insights.hasAuth ? "✓ 包含登录功能" : ""}
${analysis.insights.hasSearch ? "✓ 包含搜索功能" : ""}
${analysis.insights.isInteractive ? "✓ 页面可交互" : ""}`,
        },
      ],
      details: analysis,
    };
  },
};
```

```typescript
export const playwrightClickTool: AgentTool = {
  name: "playwright_click",
  label: "点击元素",
  description: "点击页面上的按钮、链接或其他可点击元素",

  parameters: Type.Object({
    selector: Type.String({ description: "CSS 选择器或文本内容" }),
    strategy: Type.Optional(
      Type.Union([
        Type.Literal("css"),
        Type.Literal("text"),
        Type.Literal("role"),
      ])
    ),
    waitForNavigation: Type.Optional(Type.Boolean()),
  }),

  execute: async (toolCallId, params, signal, onUpdate) => {
    const page = globalThis._testPage as Page;
    const { selector, strategy = "css", waitForNavigation = false } = params;

    let locator;
    switch (strategy) {
      case "text":
        locator = page.getByText(selector);
        break;
      case "role":
        locator = page.getByRole(selector as any);
        break;
      default:
        locator = page.locator(selector);
    }

    onUpdate?.({
      content: [{ type: "text", text: `正在点击: ${selector}` }],
      details: {},
    });

    if (waitForNavigation) {
      await Promise.all([
        page.waitForNavigation({ timeout: 10000 }),
        locator.click(),
      ]);
    } else {
      await locator.click();
    }

    // 等待页面稳定
    await page.waitForLoadState("domcontentloaded");

    return {
      content: [
        {
          type: "text",
          text: `成功点击: ${selector}\n当前URL: ${page.url()}`,
        },
      ],
      details: {
        selector,
        strategy,
        newUrl: page.url(),
      },
    };
  },
};
```

```typescript
export const playwrightAssertTool: AgentTool = {
  name: "playwright_assert",
  label: "断言检查",
  description: "验证页面元素或内容是否符合预期",

  parameters: Type.Object({
    assertType: Type.Union([
      Type.Literal("visible"),
      Type.Literal("text"),
      Type.Literal("url"),
      Type.Literal("count"),
    ]),
    selector: Type.Optional(Type.String()),
    expected: Type.String(),
  }),

  execute: async (toolCallId, params, signal, onUpdate) => {
    const page = globalThis._testPage as Page;
    const { assertType, selector, expected } = params;

    let passed = false;
    let actual: any;

    switch (assertType) {
      case "visible":
        actual = await page.locator(selector!).isVisible();
        passed = actual === (expected === "true");
        break;

      case "text":
        actual = await page.locator(selector!).textContent();
        passed = actual?.includes(expected) || false;
        break;

      case "url":
        actual = page.url();
        passed = actual.includes(expected);
        break;

      case "count":
        actual = await page.locator(selector!).count();
        passed = actual === parseInt(expected);
        break;
    }

    if (!passed) {
      throw new Error(
        `断言失败: ${assertType}\n期望: ${expected}\n实际: ${actual}`
      );
    }

    return {
      content: [
        {
          type: "text",
          text: `✓ 断言通过: ${assertType} - ${expected}`,
        },
      ],
      details: { assertType, expected, actual, passed },
    };
  },
};
```

### 2.2 测试流程编排

```typescript
// backend/services/test_agent/test_orchestrator.ts

export class TestOrchestrator {
  private agent: Agent;

  constructor() {
    this.agent = new Agent({
      initialState: {
        systemPrompt: TEST_AGENT_SYSTEM_PROMPT,
        model: getModel("anthropic", "claude-sonnet-4-20250514"),
        tools: ALL_TEST_TOOLS,
      },
      toolExecution: "parallel",
      beforeToolCall: this.beforeToolCall,
      afterToolCall: this.afterToolCall,
    });
  }

  /**
   * 执行完整的冒烟测试
   */
  async runSmokeTest(siteId: string, requirement: string): Promise<TestReport> {
    const siteUrl = `http://localhost:19100/sites/${siteId}`;

    // 订阅 Agent 事件（用于实时日志）
    this.agent.subscribe((event) => {
      this.handleAgentEvent(event);
    });

    // 启动测试流程
    const prompt = `
请对以下网站进行全面的冒烟测试：

**网站信息**
- URL: ${siteUrl}
- 需求: ${requirement}

**测试步骤**
1. 导航到网站首页
2. 分析页面结构，识别核心功能模块
3. 针对每个功能模块设计测试用例：
   - 页面加载测试
   - 交互功能测试（按钮、表单、链接）
   - 数据展示测试
   - 错误处理测试
4. 执行所有测试用例
5. 对失败的用例进行重试（最多 2 次）
6. 生成详细的测试报告

**注意事项**
- 所有关键操作都要截图
- 发现问题时要记录详细的错误信息
- 测试报告要包含建议和改进方向
`;

    await this.agent.prompt(prompt);

    // 等待测试完成
    await this.agent.waitForIdle();

    // 提取测试报告
    const report = this.extractTestReport();

    // 清理资源
    await this.cleanup();

    return report;
  }

  /**
   * 工具执行前检查
   */
  private beforeToolCall = async ({ toolCall, args }: any) => {
    // 安全检查
    if (toolCall.name === "playwright_fill") {
      const value = args.value;
      if (this.containsDangerousContent(value)) {
        return { block: true, reason: "检测到潜在的危险输入" };
      }
    }

    return undefined;
  };

  /**
   * 工具执行后处理
   */
  private afterToolCall = async ({ toolCall, result, isError }: any) => {
    if (isError) {
      // 记录错误到数据库
      await this.logError(toolCall, result);
    }

    return undefined;
  };

  /**
   * 处理 Agent 事件
   */
  private handleAgentEvent(event: any) {
    switch (event.type) {
      case "message_update":
        // 流式日志
        if (event.assistantMessageEvent.type === "text_delta") {
          this.streamLog(event.assistantMessageEvent.delta);
        }
        break;

      case "tool_execution_start":
        this.logToolStart(event.toolCallId, event.toolName, event.args);
        break;

      case "tool_execution_end":
        this.logToolEnd(event.toolCallId, event.result);
        break;

      case "agent_end":
        this.logTestComplete(event.messages);
        break;
    }
  }

  /**
   * 清理资源
   */
  private async cleanup() {
    const browser = globalThis._testBrowser;
    if (browser) {
      await browser.close();
      delete globalThis._testBrowser;
      delete globalThis._testPage;
    }
  }
}
```

```python
# backend/tasks/intelligent_test.py

from celery import Task
from backend.core.celery_app import celery_app
from backend.services.test_agent.test_orchestrator import TestOrchestrator

@celery_app.task(bind=True, max_retries=1)
def intelligent_smoke_test_task(self, task_id: str, site_id: str):
    """
    智能冒烟测试任务（使用 pi-agent 框架）

    Args:
        task_id: 任务 ID
        site_id: 站点 ID
    """
    try:
        update_task_status(task_id, TaskStatus.RUNNING)
        append_task_log(task_id, "🤖 启动智能测试 Agent...")

        # 获取站点信息
        site = get_site(site_id)
        site_data = load_site_data(site_id)

        # 创建测试编排器
        orchestrator = TestOrchestrator(task_id=task_id)

        # 执行智能测试
        append_task_log(task_id, "🧠 AI 正在分析网站功能...")
        report = orchestrator.run_smoke_test(
            site_id=site_id,
            requirement=site_data.get("requirement", "")
        )

        # 保存测试报告
        save_test_report(task_id, report)

        # 更新任务状态
        update_task_status(task_id, TaskStatus.SUCCESS, result=report)
        append_task_log(
            task_id,
            f"✅ 测试完成: {report['statistics']['passed']}/{report['statistics']['total']} 通过"
        )

        return report

    except Exception as exc:
        append_task_log(task_id, f"❌ 测试失败: {exc}", "ERROR")
        update_task_status(task_id, TaskStatus.FAILED, error=str(exc))
```

```vue
import { ref, computed } from 'vue'
import { useWebSocket } from '@/composables/useWebSocket'
import { testsAPI } from '@/api/tests'
import { ElMessage } from 'element-plus'

const props = defineProps<{
  siteId: string
}>()

const testConfig = ref({
  depth: 'standard',
  model: 'claude-sonnet-4',
  autoRetry: true,
})

const testRunning = ref(false)
const realtimeLogs = ref<any[]>([])
const latestReport = ref<any>(null)

const canTest = computed(() => !testRunning.value)

// WebSocket 连接（实时日志）
const { connect, on, disconnect } = useWebSocket('/ws')

const startIntelligentTest = async () => {
  try {
    testRunning.value = true
    realtimeLogs.value = []

    // 创建测试任务
    const response = await testsAPI.createIntelligentTest(props.siteId, testConfig.value)

    ElMessage.success('智能测试已启动')

    // 连接 WebSocket 接收实时日志
    connect()
    on(`test:${response.task_id}:log`, (log) => {
      realtimeLogs.value.push(log)
    })

    on(`test:${response.task_id}:complete`, (report) => {
      latestReport.value = report
      testRunning.value = false
      disconnect()
      ElMessage.success('测试完成')
    })

  } catch (error) {
    testRunning.value = false
    ElMessage.error('启动测试失败')
  }
}
</script>
```

## 四、核心优势

### 4.1 智能化测试

| 特性 | 传统方式 | 智能 Agent |
|------|----------|-----------|
| **用例生成** | 手动编写 | AI 自动生成 |
| **适应性** | 固定脚本 | 自适应页面变化 |
| **问题诊断** | 人工分析日志 | AI 智能诊断 |
| **覆盖率** | 有限场景 | 全场景覆盖 |
| **维护成本** | 高（每次修改都要改脚本） | 低（AI 自适应）|

### 4.2 实际应用场景

**场景 1：新网站首次测试**

```
用户创建网站 → AI 分析需求 → 自动生成 20+ 测试用例 → 执行测试 → 生成报告
```

**场景 2：网站更新后回归测试**
```
AI 对比前后版本差异 → 识别影响范围 → 重点测试变更部分 → 快速验证
```

**场景 3：问题诊断**
```
测试失败 → AI 分析错误 → 提供修复建议 → 重试验证
```

## 六、成本估算

### 6.1 API 成本

| 模型 | 输入成本 | 输出成本 | 平均单次测试成本 |
|------|----------|----------|------------------|
| Claude Sonnet 4 | $3/1M tokens | $15/1M tokens | ~$0.50 |
| GPT-4o | $2.5/1M tokens | $10/1M tokens | ~$0.40 |
| Gemini 2.0 Pro | $1.25/1M tokens | $5/1M tokens | ~$0.20 |

**月度成本估算（假设每天 100 个测试）：**
- Claude: $0.50 × 100 × 30 = **$1,500/月**
- GPT-4o: $0.40 × 100 × 30 = **$1,200/月**
- Gemini: $0.20 × 100 × 30 = **$600/月**

### 6.2 成本优化策略

1. **模型选择**：默认使用 Gemini（成本最低），复杂场景切换 Claude
2. **缓存复用**：相似网站共享测试策略
3. **批量测试**：合并多个简单网站的测试
4. **提示词优化**：减少冗余输出
